  43. /**
  44.  * Starting point for the Solr API. Represents a Solr server resource and has
  45.  * methods for pinging, adding, deleting, committing, optimizing and searching.
  46.  *
  47.  * Example Usage:
  48.  * <code>
  49.  * ...
  50.  * $solr = new Apache_Solr_Service(); //or explicitly new Apache_Solr_Service('localhost', 8180, '/solr')
  51.  *
  52.  * if ($solr->ping())
  53.  * {
  54.  *         $solr->deleteByQuery('*:*'); //deletes ALL documents - be careful :)
  55.  *
  56.  *         $document = new Apache_Solr_Document();
  57.  *         $document->id = uniqid(); //or something else suitably unique
  58.  *
  59.  *         $document->title = 'Some Title';
  60.  *         $document->content = 'Some content for this wonderful document. Blah blah blah.';
  61.  *
  62.  *         $solr->addDocument($document);     //if you're going to be adding documents in bulk using addDocuments
  63.  *                                         //with an array of documents is faster
  64.  *
  65.  *         $solr->commit(); //commit to see the deletes and the document
  66.  *         $solr->optimize(); //merges multiple segments into one
  67.  *
  68.  *         //and the one we all care about, search!
  69.  *         //any other common or custom parameters to the request handler can go in the
  70.  *         //optional 4th array argument.
  71.  *         $solr->search('content:blah', 0, 10, array('sort' => 'timestamp desc'));
  72.  * }
  73.  * ...
  74.  * </code>
  75.  *
  76.  * @todo Investigate using other HTTP clients other than file_get_contents built-in handler. Could provide performance
  77.  *  improvements when dealing with multiple requests by using HTTP's keep alive functionality

  167.     /**
  168.      * Escape a value for special query characters such as ':', '(', ')', '*', '?', etc.
  169.      *
  170.      * NOTE: inside a phrase fewer characters need escaped, use {@link Apache_Solr_Service::escapePhrase()} instead
  171.      *
  172.      * @param string $value 
  173.      * @return string 
  174.      */
  175.     static public function escape($value)
  176.     {
  177.         //list taken from http://lucene.apache.org/java/docs/queryparsersyntax.html#Escaping%20Special%20Characters
  178.         $pattern = '/(\+|-|&&|\|\||!|\(|\)|\{|}|\[|]|\^|"|~|\*|\?|:|\\\)/';
  179.         $replace = '\\\$1';
  180.  
  181.         return preg_replace($pattern, $replace, $value);
  182.     }
  183.  
  184.     /**
  185.      * Escape a value meant to be contained in a phrase for special query characters
  186.      *
  187.      * @param string $value 
  188.      * @return string 
  189.      */
  190.     static public function escapePhrase($value)
  191.     {
  192.         $pattern = '/("|\\\)/';
  193.         $replace = '\\\$1';
  194.  
  195.         return preg_replace($pattern, $replace, $value);
  196.     }

  236.     /**
  237.      * Return a valid http URL given this server's host, port and path and a provided servlet name
  238.      *
  239.      * @param string $servlet 
  240.      * @return string 
  241.      */
  242.     protected function _constructUrl($servlet, $params = array())
  243.     {
  244.         if (count($params))
  245.         {
  246.             //escape all parameters appropriately for inclusion in the query string
  247.             $escapedParams = array();
  248.  
  249.             foreach ($params as $key => $value)
  250.             {
  251.                 $escapedParams[] = urlencode($key) . '=' . urlencode($value);
  252.             }
  253.  
  254.             $queryString = $this->_queryDelimiter . implode($this->_queryStringDelimiter, $escapedParams);
  255.         }
  256.         else
  257.         {
  258.             $queryString = '';
  259.         }
  260.  
  261.         return 'http://' . $this->_host . ':' . $this->_port . $this->_path . $servlet . $queryString;
  262.     }

  529.     /**
  530.      * Call the /admin/ping servlet, can be used to quickly tell if a connection to the
  531.      * server is able to be made.
  532.      *
  533.      * @param float $timeout maximum time to wait for ping in seconds, -1 for unlimited (default is 2)
  534.      * @return float Actual time taken to ping the server, FALSE if timeout occurs
  535.      */
  536.     public function ping($timeout = 2)
  537.     {
  538.         $timeout = (float) $timeout;
  539.  
  540.         if ($timeout <= 0)
  541.         {
  542.             $timeout = -1;
  543.         }
  544.  
  545.         $start = microtime(true);
  546.  
  547.         //to prevent strict errors
  548.         $errno = 0;
  549.         $errstr = '';
  550.  
  551.         //try to connect to the host with timeout
  552.         $fp = fsockopen($this->_host, $this->_port, $errno, $errstr, $timeout);
  553.  
  554.         if ($fp)
  555.         {
  556.             //If we have a timeout set, then determine the amount of time we have left
  557.             //in the request and set the stream timeout for the write operation
  558.             if ($timeout > 0)
  559.             {
  560.                 //do the calculation
  561.                 $writeTimeout = $timeout - (microtime(true) - $start);
  562.  
  563.                 //check if we're out of time
  564.                 if ($writeTimeout <= 0)
  565.                 {
  566.                     fclose($fp);
  567.                     return false;
  568.                 }
  569.  
  570.                 //convert to microseconds and set the stream timeout
  571.                 $writeTimeoutInMicroseconds = (int) $writeTimeout * 1000000;
  572.                 stream_set_timeout($fp, 0, $writeTimeoutInMicroseconds);
  573.             }
  574.  
  575.             $request =     'HEAD ' . $this->_path . self::PING_SERVLET . ' HTTP/1.1' . "\r\n" .
  576.                         'host: ' . $this->_host . "\r\n" .
  577.                         'Connection: close' . "\r\n" .
  578.                         "\r\n";
  579.  
  580.             fwrite($fp, $request);
  581.  
  582.             //check the stream meta data to see if we timed out during the operation
  583.             $metaData = stream_get_meta_data($fp);
  584.  
  585.             if (isset($metaData['timeout']) && $metaData['timeout'])
  586.             {
  587.                 fclose($fp);
  588.                 return false;
  589.             }
  590.  
  591.  
  592.             //if we have a timeout set and have made it this far, determine the amount of time
  593.             //still remaining and set the timeout appropriately before the read operation
  594.             if ($timeout > 0)
  595.             {
  596.                 //do the calculation
  597.                 $readTimeout = $timeout - (microtime(true) - $start);
  598.  
  599.                 //check if we've run out of time
  600.                 if ($readTimeout <= 0)
  601.                 {
  602.                     fclose($fp);
  603.                     return false;
  604.                 }
  605.  
  606.                 //convert to microseconds and set the stream timeout
  607.                 $readTimeoutInMicroseconds = $readTimeout * 1000000;
  608.                 stream_set_timeout($fp, 0, $readTimeoutInMicroseconds);
  609.             }
  610.  
  611.             //at the very least we should get a response header line of
  612.             //HTTP/1.1 200 OK
  613.             $response = fread($fp, 15);
  614.  
  615.             //check the stream meta data to see if we timed out during the operation
  616.             $metaData = stream_get_meta_data($fp);
  617.             fclose($fp); //we're done with the connection - ignore the rest
  618.  
  619.             if (isset($metaData['timeout']) && $metaData['timeout'])
  620.             {
  621.                 return false;
  622.             }
  623.  
  624.             //finally, check the response header line
  625.             if ($response != 'HTTP/1.1 200 OK')
  626.             {
  627.                 return false;
  628.             }
  629.  
  630.             //we made it, return the approximate ping time
  631.             return microtime(true) - $start;
  632.         }
  633.  
  634.         //we weren't able to make a connection
  635.         return false;
  636.     }
