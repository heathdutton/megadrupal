  /**
   * Retrieves a Formatter Object for {{ field_name }}
   *
   * @return {{ formatter_class }}
   */
  public function get{{ field_name_camelized }}() {
    if (!isset($this->field_formatter['{{ field_name }}'])) {
      $this->field_formatter['{{ field_name }}'] = new {{ formatter_class }}($this->entity->get('{{ field_name }}'), {{ formatter_multivalue }});
    }

    return $this->field_formatter['{{ field_name }}'];
  }
