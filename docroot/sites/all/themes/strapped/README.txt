/**
* Form API Reference
*/



#input_size - Control the size of input fields. (xs, sm, md, lg)
http://getbootstrap.com/css/#forms-control-sizes

#placeholder - Add placeholder text to the form element.

/**
* The following parameters are used by the column_wrapper theme function.
*/
#wrapper_columns - Add column classes of the specified widths keyed by breakpoint
#wrapper_offset - Add offset classes of the specified widths keyed by breakpoint
#wrapper_push - Add push classes of the specified widths keyed by breakpoint
#wrapper_pull - Add pull classes of the specified widths keyed by breakpoint

#row_wrapper - Add a row wrapper (true, false)

#element_wrapper - (true, false)

#form_type_wrapper - Should the form element be wrapped in the form-type wrapper div (true, false)
#form_type - Type of form element (horizontal, inline, basic)
